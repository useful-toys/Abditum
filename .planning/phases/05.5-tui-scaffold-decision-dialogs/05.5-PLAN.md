2. **Implement Update() method:**
   ```go
   func (d *DecisionDialog) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
     switch msg := msg.(type) {
     case tea.KeyPressMsg:
       key := msg.String()
       
       // Handle explicit key (s, n, enter, esc, arrows)
       for i, act := range d.actions {
         if strings.EqualFold(key, act.Key) || 
            (key == "enter" && act.IsDefault) ||
            (key == "esc" && act.IsCancel) {
           // Dispatch this action
           return d, func() tea.Msg { return popModalMsg{} }
         }
       }
       
       // Handle arrow navigation (Confirmação only)
       if d.intention == IntentConfirm {
         if key == "left" && d.selectedActionIndex > 0 {
           d.selectedActionIndex--
           return d, nil
         }
         if key == "right" && d.selectedActionIndex < len(d.actions)-1 {
           d.selectedActionIndex++
           return d, nil
         }
       }
       
     case tea.WindowSizeMsg:
       // Ignore (modal is sized by root model)
     }
     return d, nil
   }
   ```
