```go
func (m *filePickerModal) View() string {
    // Nil-safe theme (D-17)
    theme := m.theme
    if theme == nil {
        theme = ThemeTokyoNight
    }

    // Layout dimensions (D-08)
    modalW := min(70, m.width*8/10)
    if modalW < 20 {
        modalW = 20
    }
    modalH := m.height * 8 / 10
    if modalH < 6 {
        modalH = 6
    }
    innerW := modalW - 2
    treeW := innerW * 40 / 100
    if treeW < 8 {
        treeW = 8
    }
    filesW := innerW - treeW - 1
    if filesW < 8 {
        filesW = 8
    }

    visibleH := modalH - 4 // top border + Caminho + panel sep + bottom border
    if m.mode == FilePickerSave {
        visibleH -= 3 // separator + field row + padding line
    }
    if visibleH < 1 {
        visibleH = 1
    }

    var lines []string

    // 1. Top border
    lines = append(lines, m.renderTopBorder(modalW, theme))

    // 2. Caminho header (D-20)
    lines = append(lines, m.renderCaminhoHeader(innerW, theme))

    // 3. Panel separator with ├─ Estrutura ──┬─ Arquivos ──┤ headers
    lines = append(lines, m.renderPanelSeparator(innerW, treeW, filesW, theme))

    // 4. Panel content rows
    treeLines  := m.renderTreeContent(treeW, visibleH, theme)
    filesLines := m.renderFilesContent(filesW, visibleH, theme)
    for i := 0; i < visibleH; i++ {
        var tl, fl string
        if i < len(treeLines) {
            tl = treeLines[i]
        }
        if i < len(filesLines) {
            fl = filesLines[i]
        }
        // Pad tree side to exactly treeW
        tl = padRight(tl, treeW)
        fl = padRight(fl, filesW)

        // Separator char: │ with scroll indicator logic (D-08)
        sep := renderTreeSepChar(m.treeScroll, len(m.visibleNodes), visibleH, i, theme)

        // Left border │, panel content, tree/files sep, right border │ with scroll indicator
        leftBorderSt := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderFocused))
        rightBorder  := renderFileSepChar(m.fileScroll, len(m.files), visibleH, i, theme)

        lines = append(lines, leftBorderSt.Render("│")+tl+sep+fl+rightBorder)
    }

    // 5. Save mode campo nome section (D-08, D-12)
    if m.mode == FilePickerSave {
        lines = append(lines, m.renderFieldSeparator(innerW, treeW, theme))
        lines = append(lines, m.renderFieldRow(innerW, theme))
    }

    // 6. Bottom border (D-08)
    lines = append(lines, m.renderBottomBorder(innerW, treeW, theme))

    return strings.Join(lines, "\n")
}
```

**padRight() helper — pads to exact visual width (ANSI-aware):**
```go
func padRight(s string, width int) string {
    w := lipgloss.Width(s)
    if w >= width {
        return s
    }
    return s + strings.Repeat(" ", width-w)
}
```

```go
func (m *filePickerModal) renderTopBorder(modalW int, theme *Theme) string {
    borderSt := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderFocused))
    titleSt  := lipgloss.NewStyle().Foreground(theme.TextPrimary).Bold(true)

    title := " " + m.title + " "
    titleRendered := titleSt.Render(title)
    // Available dashes after "╭── " and before " ───…──╮"
    // total inner = modalW - 2 (corners)
    // used: len("── ") + titleVisual + len(" ───")
    prefixDashes := "── "
    suffixPrefix := " "
    used := lipgloss.Width(prefixDashes) + lipgloss.Width(title) + lipgloss.Width(suffixPrefix)
    remaining := modalW - 2 - used
    if remaining < 1 {
        remaining = 1
    }
    dashes := strings.Repeat("─", remaining)
    return borderSt.Render("╭") + borderSt.Render(prefixDashes) + titleRendered + borderSt.Render(suffixPrefix+dashes+"╮")
}
```

```go
func (m *filePickerModal) renderCaminhoHeader(innerW int, theme *Theme) string {
    borderSt  := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderFocused))
    labelSt   := lipgloss.NewStyle().Foreground(theme.TextSecondary)
    valueSt   := lipgloss.NewStyle().Foreground(theme.TextPrimary)

    label := "Caminho: "
    path  := m.currentPath
    // Truncate path from left with … if too wide (D-11)
    avail := innerW - lipgloss.Width(label)
    if avail < 3 {
        avail = 3
    }
    if lipgloss.Width(path) > avail {
        // Truncate from left: keep suffix, prefix with …
        for lipgloss.Width("…"+path) > avail && len(path) > 1 {
            // Remove one rune from the front
            for i := 1; i <= len(path); i++ {
                if path[i-1] < 128 || path[i-1] >= 192 {
                    path = "…" + path[i:]
                    break
                }
            }
        }
    }
    content := labelSt.Render(label) + valueSt.Render(path)
    // Pad to innerW
    content = padRight(content, innerW)
    return borderSt.Render("│") + content + borderSt.Render("│")
}
```

```go
func (m *filePickerModal) renderPanelSeparator(innerW, treeW, filesW int, theme *Theme) string {
    sepSt    := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderDefault))
    headerSt := lipgloss.NewStyle().Foreground(theme.TextSecondary).Bold(true)
    borderSt := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderFocused))

    // Tree side: "─ Estrutura ─…─"
    treeLabel := " Estrutura "
    treeLabelW := lipgloss.Width(treeLabel)
    treeDashes := treeW - treeLabelW
    if treeDashes < 0 {
        treeDashes = 0
    }
    treeLeft  := strings.Repeat("─", treeDashes/2)
    treeRight := strings.Repeat("─", treeDashes-treeDashes/2)

    // Files side: "─ Arquivos ─…─"
    filesLabel := " Arquivos "
    filesLabelW := lipgloss.Width(filesLabel)
    filesDashes := filesW - filesLabelW
    if filesDashes < 0 {
        filesDashes = 0
    }
    filesLeft  := strings.Repeat("─", filesDashes/2)
    filesRight := strings.Repeat("─", filesDashes-filesDashes/2)

    return (borderSt.Render("├") +
        sepSt.Render(treeLeft) + headerSt.Render(treeLabel) + sepSt.Render(treeRight) +
        sepSt.Render("┬") +
        sepSt.Render(filesLeft) + headerSt.Render(filesLabel) + sepSt.Render(filesRight) +
        borderSt.Render("┤"))
}
```

```go
func (m *filePickerModal) renderTreeContent(treeW, visibleH int, theme *Theme) []string {
    selectedSt  := lipgloss.NewStyle().Foreground(theme.AccentPrimary).Bold(true)
    normalSt    := lipgloss.NewStyle().Foreground(theme.TextPrimary)
    indicatorSt := lipgloss.NewStyle().Foreground(theme.AccentSecondary)

    var lines []string
    end := m.treeScroll + visibleH
    if end > len(m.visibleNodes) {
        end = len(m.visibleNodes)
    }
    for i := m.treeScroll; i < end; i++ {
        node := m.visibleNodes[i].node
        indent := strings.Repeat("  ", node.depth)

        // Indicator (D-01)
        var indicator string
        if node.depth == 0 {
            indicator = "  " // root has no indicator
        } else if node.hasSubdirs == false {
            indicator = "▷ "
        } else if node.expanded {
            indicator = "▼ "
        } else {
            indicator = "▶ "
        }

        nameText := indent + node.name
        // Truncate if too wide (leave room for indicator)
        indicatorW := lipgloss.Width(indicator)
        maxNameW := treeW - indicatorW
        if maxNameW < 1 {
            maxNameW = 1
        }
        if lipgloss.Width(nameText) > maxNameW {
            nameText = nameText[:maxNameW-1] + "…"
        }

        indSt := indicatorSt
        nameSt := normalSt
        if i == m.treeCursor {
            nameSt = selectedSt
        }

        line := indSt.Render(indicator) + nameSt.Render(nameText)
        lines = append(lines, line)
    }
    // Fill remaining rows with empty
    for len(lines) < visibleH {
        lines = append(lines, "")
    }
    return lines
}
```

**renderTreeSepChar() + renderFileSepChar() — scroll indicator logic (D-08):**

```go
// renderTreeSepChar returns the character for the tree/files separator column at row i.
// The │ is replaced by ↑, ■, or ↓ for scroll indicators (D-08).
func renderTreeSepChar(scroll, total, visibleH, row int, theme *Theme) string {
    sepSt := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderDefault))
    indSt := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorTextSecondary))
    if total <= visibleH {
        return sepSt.Render("│")
    }
    if row == 0 && scroll > 0 {
        return indSt.Render("↑")
    }
    if row == visibleH-1 && scroll+visibleH < total {
        return indSt.Render("↓")
    }
    // Scrollbar thumb ■
    if total > visibleH {
        thumbPos := (scroll * (visibleH - 2)) / (total - visibleH)
        if thumbPos < 0 {
            thumbPos = 0
        }
        if thumbPos > visibleH-3 {
            thumbPos = visibleH - 3
        }
        if row == thumbPos+1 {
            return indSt.Render("■")
        }
    }
    return sepSt.Render("│")
}

// renderFileSepChar returns the right modal border char at row i (scroll indicators replace │ for files).
func renderFileSepChar(scroll, total, visibleH, row int, theme *Theme) string {
    borderSt := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderFocused))
    indSt    := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorTextSecondary))
    if total <= visibleH {
        return borderSt.Render("│")
    }
    if row == 0 && scroll > 0 {
        return indSt.Render("↑")
    }
    if row == visibleH-1 && scroll+visibleH < total {
        return indSt.Render("↓")
    }
    if total > visibleH {
        thumbPos := (scroll * (visibleH - 2)) / (total - visibleH)
        if thumbPos < 0 {
            thumbPos = 0
        }
        if thumbPos > visibleH-3 {
            thumbPos = visibleH - 3
        }
        if row == thumbPos+1 {
            return indSt.Render("■")
        }
    }
    return borderSt.Render("│")
}
```

```go
func (m *filePickerModal) renderFilesContent(filesW, visibleH int, theme *Theme) []string {
    normalSt   := lipgloss.NewStyle().Foreground(theme.TextPrimary)
    bulletSt   := lipgloss.NewStyle().Foreground(theme.TextSecondary)
    metaSt     := lipgloss.NewStyle().Foreground(theme.TextSecondary)
    disabledSt := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorTextSecondary))

    selectedBg := lipgloss.Color("#3d59a1") // special.highlight (D-09 hardcoded)
    selectedSt := lipgloss.NewStyle().
        Background(selectedBg).
        Foreground(theme.TextPrimary).
        Bold(true)

    if len(m.files) == 0 {
        // "Nenhum cofre neste diretório" centered, text.secondary (D-15)
        msg := "Nenhum cofre neste diretório"
        msgW := lipgloss.Width(msg)
        pad := (filesW - msgW) / 2
        if pad < 0 {
            pad = 0
        }
        emptyLine := strings.Repeat(" ", pad) + disabledSt.Render(msg)
        var lines []string
        emptyRow := visibleH / 2
        for i := 0; i < visibleH; i++ {
            if i == emptyRow {
                lines = append(lines, padRight(emptyLine, filesW))
            } else {
                lines = append(lines, "")
            }
        }
        return lines
    }

    // Metadata column widths for D-11 layout:
    // "● {name:<N}  {size:>8}  {datetime:14}" where 14 = len("02/01/06 15:04")
    const dateW = 14
    const sizeW = 8
    const sep   = 2 // spaces between columns
    nameW := filesW - 1 - 1 - sizeW - sep - dateW - sep // 1=bullet, 1=space
    if nameW < 4 {
        nameW = 4
    }

    var lines []string
    end := m.fileScroll + visibleH
    if end > len(m.files) {
        end = len(m.files)
    }
    for i := m.fileScroll; i < end; i++ {
        name := m.files[i]
        // Truncate filename if needed (priority: filename last — D-11)
        if lipgloss.Width(name) > nameW {
            name = name[:nameW-1] + "…"
        }

        var sizePart, datePart string
        if i < len(m.fileInfos) && m.fileInfos[i] != nil {
            sizePart = fmt.Sprintf("%*s", sizeW, formatFileSize(m.fileInfos[i].Size()))
            datePart = m.timeFmt(m.fileInfos[i].ModTime())
        }

        // Pad name to nameW
        namePadded := padRight(name, nameW)

        if i == m.fileCursor {
            // Selected row: entire row in selectedSt background (D-09)
            rowText := " " + namePadded + strings.Repeat(" ", sep) + sizePart + strings.Repeat(" ", sep) + datePart
            rowText = padRight(rowText, filesW-1) // -1 for ● bullet
            line := bulletSt.Render("●") + selectedSt.Render(rowText)
            lines = append(lines, padRight(line, filesW))
        } else {
            bullet := bulletSt.Render("●")
            nameR  := normalSt.Render(" " + namePadded)
            sizeR  := metaSt.Render(strings.Repeat(" ", sep) + sizePart)
            dateR  := metaSt.Render(strings.Repeat(" ", sep) + datePart)
            line   := bullet + nameR + sizeR + dateR
            lines  = append(lines, padRight(line, filesW))
        }
    }
    // Fill remaining
    for len(lines) < visibleH {
        lines = append(lines, "")
    }
    return lines
}
```

```go
        sepSt.Render("┴") +
```

```go
func (m *filePickerModal) renderFieldRow(innerW int, theme *Theme) string {
    borderSt := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderFocused))

    isFocused := m.focusPanel == 2
    var labelSt lipgloss.Style
    if isFocused {
        labelSt = lipgloss.NewStyle().Foreground(theme.AccentPrimary).Bold(true)
    } else {
        labelSt = lipgloss.NewStyle().Foreground(theme.TextSecondary)
    }
    fieldBgSt := lipgloss.NewStyle().Background(lipgloss.Color(ColorSurfaceInput))

    label := labelSt.Render("Nome do arquivo") + " "
    labelW := lipgloss.Width("Nome do arquivo ")

    // Field content: show value + cursor (▌ if focused)
    val := m.nameField.Value()
    fieldW := innerW - labelW - 2 // -2 for margins
    if fieldW < 4 {
        fieldW = 4
    }
    var fieldContent string
    if isFocused {
        cursor := lipgloss.NewStyle().Foreground(theme.TextPrimary).Render("▌")
        fieldContent = val + cursor
    } else {
        fieldContent = val
    }
    // Truncate if wider than fieldW
    if lipgloss.Width(fieldContent) > fieldW {
        // Keep rightmost part (most recent input)
        fieldContent = "…" + fieldContent[lipgloss.Width(fieldContent)-fieldW+1:]
    }
    fieldRendered := fieldBgSt.Render(padRight(" "+fieldContent+" ", fieldW+2))

    content := label + fieldRendered
    content = padRight(content, innerW)
    return borderSt.Render("│") + content + borderSt.Render("│")
}
```

```go
func (m *filePickerModal) renderBottomBorder(innerW, treeW int, theme *Theme) string {
    borderSt   := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderFocused))
    sepSt      := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderDefault))
    activeSt   := lipgloss.NewStyle().Foreground(theme.AccentPrimary).Bold(true)
    disabledSt := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorTextDisabled))
    cancelSt   := lipgloss.NewStyle().Foreground(theme.TextPrimary)

    // Determine active/blocked state (D-08)
    var actionActive bool
    if m.mode == FilePickerOpen {
        actionActive = len(m.files) > 0 // active if dir has files (even if focus is on tree)
    } else {
        actionActive = m.nameField.Value() != ""
    }

    var actionLabel string
    if m.mode == FilePickerOpen {
        actionLabel = "Enter Abrir"
    } else {
        actionLabel = "Enter Salvar"
    }
    var actionSt lipgloss.Style
    if actionActive {
        actionSt = activeSt
    } else {
        actionSt = disabledSt
    }

    cancelLabel := "Esc Cancelar"

    // Layout: ╰── {action} ──…──[┴]──…── {cancel} ──╯
    // In Open mode: ┴ at treeW+1 position from inner left
    actionRendered := " " + actionSt.Render(actionLabel) + " "
    cancelRendered := " " + cancelSt.Render(cancelLabel) + " "
    actionW := lipgloss.Width(actionRendered)
    cancelW := lipgloss.Width(cancelRendered)

    // Available fill: innerW - actionW - cancelW - left "── " - right " ──"
    // Let's build it by filling the middle with ─ chars
    const sideLen = 2 // "── " = 2 dashes + 1 space counted in actionRendered
    fillW := innerW - actionW - cancelW - 2 // 2 for each end ── prefix
    if fillW < 0 {
        fillW = 0
    }

    var mid string
    if m.mode == FilePickerOpen {
        // Insert ┴ at treeW+1 position in the fill zone
        leftFill := treeW - 2 // offset from start of fill (approximate)
        if leftFill < 0 {
            leftFill = 0
        }
        rightFill := fillW - leftFill - 1
        if rightFill < 0 {
            rightFill = 0
        }
        mid = sepSt.Render(strings.Repeat("─", leftFill)+"┴"+strings.Repeat("─", rightFill))
    } else {
        mid = sepSt.Render(strings.Repeat("─", fillW))
    }

    return (borderSt.Render("╰") +
        sepSt.Render("── ") +
        actionRendered +
        mid +
        cancelRendered +
        sepSt.Render(" ──") +
        borderSt.Render("╯"))
}
```
