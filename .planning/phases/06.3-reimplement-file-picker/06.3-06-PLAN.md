```go
// ANTES:
func (m *filePickerModal) renderCaminhoHeader(innerW int, theme *Theme) string {
    borderSt := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderFocused))
    labelSt := lipgloss.NewStyle().Foreground(theme.TextSecondary)
    valueSt := lipgloss.NewStyle().Foreground(theme.TextPrimary)

    label := "Caminho: "
    path := m.currentPath
    avail := innerW - lipgloss.Width(label)
    if avail < 3 {
        avail = 3
    }
    // Truncate path from left with … if too wide
    runes := []rune(path)
    for len(runes) > avail {
        runes = runes[1:]
    }
    if string(runes) != path {
        path = "…" + string(runes)
    }
    content := labelSt.Render(label) + valueSt.Render(path)
    content = padRight(content, innerW)
    return borderSt.Render("│") + content + borderSt.Render("│")
}

// DEPOIS:
func (m *filePickerModal) renderCaminhoHeader(innerW int, theme *Theme) string {
    borderSt := lipgloss.NewStyle().Foreground(lipgloss.Color(ColorBorderFocused))
    valueSt := lipgloss.NewStyle().Foreground(theme.TextPrimary)

    // 1 espaço de padding em cada lado; path usa o espaço restante
    padding := 1
    avail := innerW - padding*2
    if avail < 3 {
        avail = 3
    }
    path := m.currentPath
    // Truncate path from left with … if too wide
    runes := []rune(path)
    for len(runes) > avail {
        runes = runes[1:]
    }
    if string(runes) != path {
        path = "…" + string(runes)
    }
    content := strings.Repeat(" ", padding) + valueSt.Render(path)
    content = padRight(content, innerW)
    return borderSt.Render("│") + content + borderSt.Render("│")
}
```

**Correção 3 — renderPanelSeparator: verificar alinhamento**

A lógica atual de `treeW + 1(┬) + filesW = innerW` está correta na teoria.
Mas verificar se `filesW = innerW - treeW - 1` está sendo usado corretamente:
- `treeW = innerW * 40 / 100` (truncado por divisão inteira)
- `filesW = innerW - treeW - 1`
- Soma: `treeW + 1 + filesW = treeW + 1 + innerW - treeW - 1 = innerW` ✓

O problema pode ser que os dashes são calculados como:
```
treeLeft + treeLabel + treeRight = treeW
filesLeft + filesLabel + filesRight = filesW
```
Verificar se `treeDashes = treeW - treeLabelW` usa `lipgloss.Width(treeLabel)` ou `len(treeLabel)`.
Ambos devem dar o mesmo resultado para ASCII. Se não há bug aqui, o desalinhamento
nos goldens é puramente visual no terminal (bytes de controle ANSI que o `Get-Content`
renderiza mal). Confirmar rodando o teste e lendo o arquivo — se `go test` passa, o
alinhamento está correto no runtime.
